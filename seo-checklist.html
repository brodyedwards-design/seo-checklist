    const storageKey = "cc-seo-checklist-v1";

    const openKey = "cc-seo-checklist-open-v1";

    const saved = JSON.parse(localStorage.getItem(storageKey) || "{}");

    const openSaved = JSON.parse(localStorage.getItem(openKey) || "{}");

    const sectionsEl = document.getElementById("sections");

    const progressBar = document.getElementById("progressBar");

    const scoreText = document.getElementById("scoreText");

    function render() {

      sectionsEl.innerHTML = "";

      data.forEach((section, sIndex) => {

        const sectionId = `section-${sIndex}`;

        const isOpen = openSaved[sectionId] ?? (sIndex < 2);

        const sectionEl = document.createElement("div");

        sectionEl.className = `section ${isOpen ? "open" : ""}`;

        sectionEl.dataset.sectionId = sectionId;

        const header = document.createElement("button");

        header.className = "section-header";

        header.innerHTML = `<span>${section.title}</span><span>${isOpen ? "−" : "+"}</span>`;

        header.addEventListener("click", () => {

          const currentlyOpen = sectionEl.classList.contains("open");

          sectionEl.classList.toggle("open", !currentlyOpen);

          openSaved[sectionId] = !currentlyOpen;

          localStorage.setItem(openKey, JSON.stringify(openSaved));

          header.innerHTML = `<span>${section.title}</span><span>${!currentlyOpen ? "−" : "+"}</span>`;

        });

        const body = document.createElement("div");

        body.className = "section-body";

        if (section.note) {

          const note = document.createElement("p");

          note.className = "note";

          note.textContent = section.note;

          body.appendChild(note);

        }

        const checklist = document.createElement("div");

        checklist.className = "checklist";

        section.items.forEach((item, iIndex) => {

          const itemId = `${sectionId}-item-${iIndex}`;

          const checked = !!saved[itemId];

          const label = document.createElement("label");

          label.className = `item ${checked ? "done" : ""}`;

          const checkbox = document.createElement("input");

          checkbox.type = "checkbox";

          checkbox.checked = checked;

          checkbox.addEventListener("change", () => {

            saved[itemId] = checkbox.checked;

            localStorage.setItem(storageKey, JSON.stringify(saved));

            label.classList.toggle("done", checkbox.checked);

            updateProgress();

          });

          const text = document.createElement("div");

          text.className = "item-text";

          text.textContent = item;

          label.appendChild(checkbox);

          label.appendChild(text);

          checklist.appendChild(label);

        });

        body.appendChild(checklist);

        sectionEl.appendChild(header);

        sectionEl.appendChild(body);

        sectionsEl.appendChild(sectionEl);

      });

      updateProgress();

    }

    function updateProgress() {

      const total = data.reduce((sum, s) => sum + s.items.length, 0);

      let done = 0;

      data.forEach((section, sIndex) => {

        section.items.forEach((_, iIndex) => {

          const itemId = `section-${sIndex}-item-${iIndex}`;

          if (saved[itemId]) done++;

        });

      });

      const pct = total ? Math.round((done / total) * 100) : 0;

      progressBar.style.width = pct + "%";

      scoreText.textContent = `${done} / ${total} complete`;

    }

    document.getElementById("expandAll").addEventListener("click", () => {

      data.forEach((_, sIndex) => openSaved[`section-${sIndex}`] = true);

      localStorage.setItem(openKey, JSON.stringify(openSaved));

      render();

    });

    document.getElementById("collapseAll").addEventListener("click", () => {

      data.forEach((_, sIndex) => openSaved[`section-${sIndex}`] = false);

      localStorage.setItem(openKey, JSON.stringify(openSaved));

      render();

    });

    document.getElementById("resetAll").addEventListener("click", () => {

      if (!confirm("Reset all checklist progress?")) return;

      localStorage.removeItem(storageKey);

      localStorage.removeItem(openKey);

      Object.keys(saved).forEach(k => delete saved[k]);

      Object.keys(openSaved).forEach(k => delete openSaved[k]);

      render();

    });

    render();
